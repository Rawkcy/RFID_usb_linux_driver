00023 
00024   unsigned int ret = USB_MATCH_NONE;
00025 
00026   TRACE("comparing match specifications to USB device...");
00027 
00028   struct usb_device const* dev = usb_device((usb_dev_handle*)dev_h);
00029   
00030   TRACE("inspecting vendor ID...");
00031   if (dev->descriptor.idVendor > 0 &&
00032       (dev->descriptor.idVendor & match->vendor_id) == match->vendor_id) {
00033       TRACE("match on vendor ID: 0x%04x.", dev->descriptor.idVendor);
00034       ret |= USB_MATCH_VENDOR;

00036   else TRACE("no match on vendor ID.");
00037 
00038   TRACE("inspecting product ID...");
00039   if ((dev->descriptor.idProduct & match->product_id) == match->product_id) {
00040       TRACE("match on product ID: 0x%04x.", dev->descriptor.idProduct);
00041       ret |= USB_MATCH_PRODUCT;
00042   }
00043   else TRACE("no match on product ID.");
00044 
00045   if (match->matcher_fn) {
00046     TRACE("calling custom matching function...");
00047     if ((*match->matcher_fn)(dev_h, match->custom_data, match->custom_data_length)) {
00048       TRACE("match on custom matching function.");
00049       ret |= USB_MATCH_CUSTOM;
00050     }
00051     else TRACE("no match on custom matching function.");
00052   }
00053   else {
00054     TRACE("no custom matching function supplied.");
00055     ret |= USB_MATCH_CUSTOM;

00061 static hid_return hid_find_usb_device(HIDInterface* const hidif,
00062     HIDInterfaceMatcher const* const match)
00063 {
00064   ASSERT(!hid_is_opened(hidif));
00065   ASSERT(match);
00066 
00067   /* WARNING: not thread-safe. usb_busses in libusb/usb.c is a global
00068    * variable.
00069    */
00070   struct usb_bus *usbbus = usb_get_busses();
00071   struct usb_device *usbdev;
00072 
00073   TRACE("enumerating USB busses...");
00074   for (; usbbus; usbbus = usbbus->next) {
00075 
00076     TRACE("enumerating USB devices on bus %s...", usbbus->dirname);
00077     for (usbdev = usbbus->devices; usbdev; usbdev=usbdev->next) {
00078 
00079       snprintf(hidif->id, sizeof(hidif->id), "%s/%s[%d]",
00080           usbbus->dirname, usbdev->filename, hidif->interface);
00081 
00082       TRACE("inspecting USB device %s...", hidif->id);
00083       usb_dev_handle *usbdev_h = usb_open(usbdev);
00084 
00085       if (usbdev_h) {
00086         usb_claim_interface(usbdev_h, hidif->interface);
00087 
00088         unsigned int flags = hid_compare_usb_device(usbdev_h, match);
00089         if (flags == USB_MATCH_ALL) {
00090           NOTICE("found a matching USB device %s.", hidif->id);
00091           hidif->dev_handle = usbdev_h;
00092           hidif->device = usb_device(usbdev_h);
00093           return HID_RET_SUCCESS;
00094         }
00095 
00096         if (!(flags & USB_MATCH_VENDOR)) {
00097           NOTICE("vendor 0x%04x of USB device %s does not match 0x%04x.",
00098              usbdev->descriptor.idVendor, hidif->id, match->vendor_id);
00099         }
00100         else if (!(flags & USB_MATCH_PRODUCT)) {
00101           NOTICE("product 0x%04x of USB device %s does not match 0x%04x.",
00102              usbdev->descriptor.idProduct, hidif->id, match->product_id);
00103         }
00104         else if (!(flags & USB_MATCH_CUSTOM)) {
00105           NOTICE("custom matching function returned false on %s.", hidif->id);
00106         }
00107         //usb_release_interface(usbdev_h, hidif->interface);
00108         usb_close(usbdev_h);
00109       }
00110       else {
00111         ERROR("failed to open USB device %s", hidif->id);
00112         return HID_RET_FAIL_OPEN_DEVICE;
00113       }
00114     }
00115   }
00116   WARNING("no matching USB device found.");
00117   return HID_RET_DEVICE_NOT_FOUND;
